        class Solution {
            public int mySqrt(int x) {
                String s = String.valueOf(x);
                if(x <= 1)return x;
                int l = (s.length()/2 - 1) * 10;
                int r = 46340;
        
                while(l< r){
                    int mid = l + (r-l)/2;
                    int val = mid*mid;
                    if(x < val){
                        r = mid -1;
                    }else if(x > val){
                        l = mid + 1;
                    }else return mid;
                }
                
                return l*l <= x?l:l-1;
            }
        }
